resize : set you need size and data name
exsample: resize:{"b":1024,"n":460,"m":200,"s":50}

preview_size: preview pic size, need is number.
example: preview_size:100

drop_info: drop area show text

select_bt: select pic button.
example:select_bt:'<input type="button" value="Select Files" in_option/>'
        or
        select_bt:'<div class='select_bt' > select </div>'

submit_bt: select pic button.
example:submit_bt:'<input type="button" value="Upload to server" in_option/>'
        or
        submit_bt:'<div class='select_bt' > Upload to server </div>'

loading_info:when process,show Loading UI.
example:loading_info:'<img src="/images/loader_128x15.gif">'

filter:need upload file type.
example:filter : /image/i

ajax_type: upload to server method.
example: ajax_type:'POST'

ajax_url: server url.
example: ajax_url:'upload.php'

ajax_data: Extended input data to server.
example:ajax_data:{lib:100,pageid:100}

ajax_success: when one file upload finish callback.
example:ajax_success:function(server_return_data){
                          alert(server_return_data);
                     }

ajax_finish:  when all files upload finish callback.
example:ajax_finish:function(){
                          alert("finish");
                    }

delay:set time out delay time, need number.

lang:custom process info laguage or button show name.
example:lang:{selectBTShowName:"選擇檔案",
              init_error:"初始化錯誤"}

multiple:single file or multiple files.
example:multiple:true

process_info:show process info.
example:process_info:true
